def calc(a, b, op):
    result = { '+':a+b,
               '-':a-b,
               '*':a*b,
               '/':a/b}
    return result[op]

print(calc(10,20,'+'))

def gugudan(dan_start, dan_finish, lines):
    for line in range(1,lines+1,1):
        str=''
        for dan in range (dan_start, dan_finish+1,1):
            str = str+"%dx%d=%d\t"%(dan, line, dan*line)
        print(str)
gugudan(2,5,2)
gugudan(7,10,4)

def calc(a,b,strOP):
    #a,b - value
    #strOP - '+', '-', '*', '/'
    ret = 0
   
    if strOP == '+': ret = a+b
    if strOP == '-': ret = a-b
    if strOP == '*': ret = a*b
    if strOP == '/': ret = a/b
    
    return ret
